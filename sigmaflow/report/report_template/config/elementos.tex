% -------- CAPTION GERAL --------
\captionsetup{
    labelsep=endash,
    justification=justified,
    singlelinecheck=false,
    font=normalsize,
    labelfont=bf
}

% -------- FIGURA: legenda centralizada --------
\captionsetup[figure]{
    position=below,
    justification=centering,
    singlelinecheck=false
}

% -------- TABELA: legenda acima --------
\captionsetup[table]{
    position=above,
    justification=centering,
    singlelinecheck=false
}

% -------- QUADRO: contador pr?prio + legenda acima --------
\newcounter{quadro}
\newcommand{\quadrocaption}[1]{%
    \refstepcounter{quadro}%
    \par\noindent\textbf{Quadro \thequadro} -- #1\par%
}

% -------- FONTE: linha abaixo de figuras, tabelas e quadros --------
% Uso: \fonte{Autor (ano)}  ou  \fonte{SigmaFlow (ano)}
\newcommand{\fonte}[1]{%
    \par\noindent Fonte: #1\par%
}
